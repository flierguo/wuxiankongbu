如何脚本刷怪
export function Main(Npc: TNormNpc, Player: TPlayObject, Args: TArgs): void {
    //普通刷怪
    GameLib.MonGen('0', '鸡', 10, 300, 300, 5, 0, 0);   //在地图“0”的X坐标300和Y坐标300的位置上的5个坐标范围内刷10只鸡,阵营和国别均为0(无阵营无国别) 
    //类似怪物攻城
    GameLib.MobPlace('0', 300, 300, '白野猪', 100, 100, 10, 5, 0, 0);
    //在地图"0"的坐标100、100的为中心的位置上，坐标偏移5范围内，刷新10只白野猪，白野猪刷新后自动向坐标300、300上靠拢,阵营和国别均为0(无阵营无国别)
}
