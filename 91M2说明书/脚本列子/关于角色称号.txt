// 设置当前称号
export function Main(Npc: TNormNpc, Player: TPlayObject, Args: TArgs): void {
    Player.AddTitle('初入江湖'); //添加一个称号
    Player.SetActiveTitle(''); //设置当前称号为空
    Player.SetActiveTitle('初入江湖5'); //将称号"初入江湖5"设置为当前称号
    Player.RemoveTitle('初入江湖1');  //删除一个称号 
} 

// 添加一个称号
export function Main(Npc: TNormNpc, Player: TPlayObject, Args: TArgs): void {
    //AddTitle函数执行时会判断是否已经拥有该称号,如果没拥有才可以添加
    if (Player.AddTitle('初入江湖')) {
        Player.MessageBox('获得了称号“初入江湖”');
    }
}
