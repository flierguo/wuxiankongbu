如何关闭一个打开的自定义窗口
// 如何关闭一个打开的自定义窗口
export function Main(Npc: TNormNpc, Player: TPlayObject, Args: TArgs): void {
    Player.CloseWindow('Npc大窗口');    
    //执行CloseWindow后,如果客户端开启了UI里面的"Npc大窗口"则会关闭该窗口
    //UI设计里设置窗口为手动关闭后,配合此函数可以达到按照特定的条件显示和关闭一个自定义窗口
} 

// 如何更新自定义窗口上的内容
export function OnWinUpdate(Npc: TNormNpc, Player: TPlayObject, Args: TArgs): void {
    if (Player.CustomItemCount = 0)
        Player.UpdateSay('宝石进阶', '请放入需要进阶的宝石')
    else
        Player.UpdateSay('宝石进阶', '点击确定按钮执行宝石进阶');
}
