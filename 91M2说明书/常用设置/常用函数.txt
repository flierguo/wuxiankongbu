//第一种 indexOf是用于返回[目标字符串]在[源字符串]中的位置。
1、indexOf：返回特定子字符串第一次在源字符串中的位置。如果源字符中不存在目标字符，则返回-1。
//indexOf函数两种用法和实现功能说明
1、indexOf(String str): 返回指定字符str在字符串中（方法调用者）第一次出现处的起始索引，如果此字符串中没有这样的字符，则返回 -1。
2、indexOf(String str, int index): 返回从 index 位置开始查找指定字符str在字符串中第一次出现处的起始索引，如果此字符串中没有这样的字符，则返回 -1。

//第二种 includes 可以判断一个数组中是否包含某一个元素，并返回true 或者false
示例1：
['a','b','c'].includes('a') 返回 true
['a','b','c'].includes(1)   返回 false
//includes可以包含两个参数，第二个参数表示判断的起始位置
//起始位置第一个数字是0。
['a','b','c'].includes('b',2)  返回 false
['a','b','c'].includes('b',1)  返回 true
// 字符的asiic编码是否在指定范围
 function CharCodeIn(src:string,low:string,high:string):number;

// 比方如果你要过滤掉字符串中的所有阿拉伯数字 就可以这么写
 function DeleteAllNumber(str: string) : string {
    let result = '';
    for(let i = 0; i < str.length - 1; i++) {
        if(!CharCodeIn(str[i],'O','9') {
          result += str[i]
        }
    }
    return result
}

// 注意：
// 但是要注意 你不能写 CharCodeIn(str[i],'9','0')
// 因为在asiic码表里头 9 对应的代码是39
// 0 是30 这样写就有问题
